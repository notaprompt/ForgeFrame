# Cipher V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an autonomous prediction market trading agent for Kalshi that scores markets across 5 signal layers, pitches trades via push notification, executes on approval, and learns from every outcome.

**Architecture:** Standalone Python CLI. SQLite for state. Kalshi REST API for market data + execution. Kimi K2.6 (Moonshot) via OpenAI-compatible SDK for LLM reasoning. ntfy.sh for pitch delivery. Async event loop for daemon mode.

**Tech Stack:** Python 3.12+, SQLite, openai SDK (pointed at Moonshot API), httpx, numpy, pandas, statsmodels, click, ntfy.sh

```python
# cipher/config.py
"""Configuration for Cipher trading agent."""
from __future__ import annotations

import os
from dataclasses import dataclass, field
from pathlib import Path


@dataclass
class SignalWeights:
    """Current signal weights. Updated by retro engine."""
    data_disparity: float = 0.30
    sentiment_bias: float = 0.20
    rate_of_change: float = 0.15
    news_timing: float = 0.20
    structural: float = 0.15

    def as_dict(self) -> dict[str, float]:
        return {
            "data_disparity": self.data_disparity,
            "sentiment_bias": self.sentiment_bias,
            "rate_of_change": self.rate_of_change,
            "news_timing": self.news_timing,
            "structural": self.structural,
        }

    def normalize(self) -> None:
        """Ensure weights sum to 1.0."""
        total = (
            self.data_disparity
            + self.sentiment_bias
            + self.rate_of_change
            + self.news_timing
            + self.structural
        )
        if total == 0:
            return
        self.data_disparity /= total
        self.sentiment_bias /= total
        self.rate_of_change /= total
        self.news_timing /= total
        self.structural /= total


@dataclass
class Config:
    """Global configuration."""
    # Paths
    db_path: Path = field(default_factory=lambda: Path(
        os.environ.get("CIPHER_DB_PATH", "~/.cipher/cipher.db")
    ).expanduser())

    # API keys
    kalshi_api_key_id: str = field(default_factory=lambda: os.environ.get("KALSHI_API_KEY_ID", ""))
    kalshi_rsa_key_path: Path = field(default_factory=lambda: Path(
        os.environ.get("KALSHI_RSA_PRIVATE_KEY_PATH", "~/.cipher/kalshi-private-key.pem")
    ).expanduser())
    moonshot_api_key: str = field(default_factory=lambda: os.environ.get("MOONSHOT_API_KEY", ""))
    ntfy_topic: str = field(default_factory=lambda: os.environ.get("NTFY_TOPIC", "cipher-trades"))

    # Bankroll
    starting_bankroll: float = 10.00
    max_position_pct: float = 0.20  # 20% of bankroll per trade
    max_portfolio_exposure_pct: float = 0.60  # 60% total exposure
    max_open_positions: int = 3

    # Decision thresholds
    min_confidence: float = 0.50
    min_signals_agree: int = 2
    max_contradicting_signal: float = 0.70

    # Skim settings
    skim_threshold: float = 20.00  # bankroll must exceed this
    skim_rate: float = 0.25  # withdraw 25% of profits above threshold

    # Retro
    meta_retro_interval: int = 20  # every N resolved trades

    # Polling
    poll_interval_seconds: int = 60
    pitch_expiry_minutes: int = 30

    # Signal weights
    weights: SignalWeights = field(default_factory=SignalWeights)

    # Kalshi
    kalshi_base_url: str = "https://api.elections.kalshi.com/trade-api/v2"

    @classmethod
    def load(cls) -> Config:
        """Load config from environment. Call after dotenv if using one."""
        return cls()


def ensure_data_dir(config: Config) -> None:
    """Create ~/.cipher/ if it doesn't exist."""
    config.db_path.parent.mkdir(parents=True, exist_ok=True)
```

```python
# cipher/db.py
"""SQLite database schema, connection, and queries."""
from __future__ import annotations

import json
import sqlite3
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

SCHEMA_SQL = """
CREATE TABLE IF NOT EXISTS markets (
    id TEXT PRIMARY KEY,
    ticker TEXT NOT NULL,
    title TEXT NOT NULL,
    category TEXT,
    close_time TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'active',
    last_price INTEGER,
    volume INTEGER DEFAULT 0,
    yes_bid INTEGER,
    yes_ask INTEGER,
    no_bid INTEGER,
    no_ask INTEGER,
    polled_at TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS candles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    market_id TEXT NOT NULL REFERENCES markets(id),
    timestamp TEXT NOT NULL,
    open INTEGER NOT NULL,
    high INTEGER NOT NULL,
    low INTEGER NOT NULL,
    close INTEGER NOT NULL,
    volume INTEGER DEFAULT 0,
    UNIQUE(market_id, timestamp)
);

CREATE TABLE IF NOT EXISTS signals (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    market_id TEXT NOT NULL REFERENCES markets(id),
    timestamp TEXT NOT NULL,
    data_disparity REAL,
    data_disparity_conf REAL,
    sentiment_bias REAL,
    sentiment_bias_conf REAL,
    rate_of_change REAL,
    rate_of_change_conf REAL,
    news_timing REAL,
    news_timing_conf REAL,
    structural REAL,
    structural_conf REAL,
    combined_score REAL,
    confidence REAL,
    direction TEXT,
    reasoning TEXT,
    action TEXT DEFAULT 'SKIP'
);

CREATE TABLE IF NOT EXISTS trades (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    signal_id INTEGER NOT NULL REFERENCES signals(id),
    market_id TEXT NOT NULL REFERENCES markets(id),
    timestamp TEXT NOT NULL,
    direction TEXT NOT NULL,
    contracts INTEGER NOT NULL,
    price_paid INTEGER NOT NULL,
    fees REAL DEFAULT 0.0,
    resolved_at TEXT,
    outcome TEXT DEFAULT 'PENDING',
    payout REAL,
    profit_loss REAL,
    bankroll_before REAL NOT NULL,
    bankroll_after REAL
);

CREATE TABLE IF NOT EXISTS retros (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    trade_id INTEGER REFERENCES trades(id),
    market_id TEXT REFERENCES markets(id),
    resolved_at TEXT NOT NULL,
    data_was_right INTEGER,
    sentiment_was_right INTEGER,
    roc_was_right INTEGER,
    news_was_right INTEGER,
    structural_was_right INTEGER,
    prediction_error REAL,
    lesson TEXT,
    bias_detected TEXT,
    weight_adjustment TEXT,
    retro_type TEXT NOT NULL DEFAULT 'normal',
    category TEXT
);

CREATE TABLE IF NOT EXISTS denials (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    signal_id INTEGER NOT NULL REFERENCES signals(id),
    market_id TEXT NOT NULL REFERENCES markets(id),
    denied_at TEXT NOT NULL,
    would_have_won INTEGER,
    would_have_profit REAL
);

CREATE INDEX IF NOT EXISTS idx_candles_market ON candles(market_id, timestamp);
CREATE INDEX IF NOT EXISTS idx_signals_market ON signals(market_id);
CREATE INDEX IF NOT EXISTS idx_trades_market ON trades(market_id);
CREATE INDEX IF NOT EXISTS idx_trades_outcome ON trades(outcome);
CREATE INDEX IF NOT EXISTS idx_retros_type ON retros(retro_type);
CREATE INDEX IF NOT EXISTS idx_denials_market ON denials(market_id);
"""


def get_connection(db_path: Path) -> sqlite3.Connection:
    """Open a connection with row factory and WAL mode."""
    conn = sqlite3.connect(str(db_path))
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA foreign_keys=ON")
    return conn


def init_db(db_path: Path) -> sqlite3.Connection:
    """Create tables and return connection."""
    db_path.parent.mkdir(parents=True, exist_ok=True)
    conn = get_connection(db_path)
    conn.executescript(SCHEMA_SQL)
    conn.commit()
    return conn


def now_iso() -> str:
    """UTC ISO timestamp."""
    return datetime.now(timezone.utc).isoformat()


# --- Markets ---

def upsert_market(conn: sqlite3.Connection, market: dict[str, Any]) -> None:
    """Insert or update a market record."""
    conn.execute(
        """INSERT INTO markets (id, ticker, title, category, close_time, status,
           last_price, volume, yes_bid, yes_ask, no_bid, no_ask, polled_at)
           VALUES (:id, :ticker, :title, :category, :close_time, :status,
           :last_price, :volume, :yes_bid, :yes_ask, :no_bid, :no_ask, :polled_at)
           ON CONFLICT(id) DO UPDATE SET
           ticker=excluded.ticker, title=excluded.title, category=excluded.category,
           close_time=excluded.close_time, status=excluded.status,
           last_price=excluded.last_price, volume=excluded.volume,
           yes_bid=excluded.yes_bid, yes_ask=excluded.yes_ask,
           no_bid=excluded.no_bid, no_ask=excluded.no_ask,
           polled_at=excluded.polled_at""",
        market,
    )
    conn.commit()


def get_active_markets(conn: sqlite3.Connection) -> list[sqlite3.Row]:
    """Get all active markets."""
    return conn.execute(
        "SELECT * FROM markets WHERE status = 'active' ORDER BY close_time"
    ).fetchall()


def get_market(conn: sqlite3.Connection, market_id: str) -> sqlite3.Row | None:
    """Get a single market by ID."""
    return conn.execute("SELECT * FROM markets WHERE id = ?", (market_id,)).fetchone()


# --- Candles ---

def insert_candles(conn: sqlite3.Connection, candles: list[dict[str, Any]]) -> None:
    """Batch insert candles, ignoring duplicates."""
    conn.executemany(
        """INSERT OR IGNORE INTO candles
           (market_id, timestamp, open, high, low, close, volume)
           VALUES (:market_id, :timestamp, :open, :high, :low, :close, :volume)""",
        candles,
    )
    conn.commit()


def get_candles(
    conn: sqlite3.Connection, market_id: str, limit: int = 60
) -> list[sqlite3.Row]:
    """Get recent candles for a market, newest first."""
    return conn.execute(
        """SELECT * FROM candles WHERE market_id = ?
           ORDER BY timestamp DESC LIMIT ?""",
        (market_id, limit),
    ).fetchall()


# --- Signals ---

def insert_signal(conn: sqlite3.Connection, signal: dict[str, Any]) -> int:
    """Insert a signal record and return its ID."""
    cur = conn.execute(
        """INSERT INTO signals
           (market_id, timestamp, data_disparity, data_disparity_conf,
            sentiment_bias, sentiment_bias_conf, rate_of_change, rate_of_change_conf,
            news_timing, news_timing_conf, structural, structural_conf,
            combined_score, confidence, direction, reasoning, action)
           VALUES (:market_id, :timestamp, :data_disparity, :data_disparity_conf,
            :sentiment_bias, :sentiment_bias_conf, :rate_of_change, :rate_of_change_conf,
            :news_timing, :news_timing_conf, :structural, :structural_conf,
            :combined_score, :confidence, :direction, :reasoning, :action)""",
        signal,
    )
    conn.commit()
    return cur.lastrowid


def get_signal(conn: sqlite3.Connection, signal_id: int) -> sqlite3.Row | None:
    return conn.execute("SELECT * FROM signals WHERE id = ?", (signal_id,)).fetchone()


def get_pending_pitch(conn: sqlite3.Connection) -> sqlite3.Row | None:
    """Get the most recent PITCH-action signal that hasn't been approved/denied/expired."""
    return conn.execute(
        """SELECT * FROM signals WHERE action = 'PITCH'
           ORDER BY timestamp DESC LIMIT 1"""
    ).fetchone()


def update_signal_action(conn: sqlite3.Connection, signal_id: int, action: str) -> None:
    conn.execute("UPDATE signals SET action = ? WHERE id = ?", (action, signal_id))
    conn.commit()


# --- Trades ---

def insert_trade(conn: sqlite3.Connection, trade: dict[str, Any]) -> int:
    cur = conn.execute(
        """INSERT INTO trades
           (signal_id, market_id, timestamp, direction, contracts, price_paid,
            fees, outcome, bankroll_before)
           VALUES (:signal_id, :market_id, :timestamp, :direction, :contracts,
            :price_paid, :fees, :outcome, :bankroll_before)""",
        trade,
    )
    conn.commit()
    return cur.lastrowid


def get_open_trades(conn: sqlite3.Connection) -> list[sqlite3.Row]:
    return conn.execute(
        "SELECT * FROM trades WHERE outcome = 'PENDING'"
    ).fetchall()


def get_trade(conn: sqlite3.Connection, trade_id: int) -> sqlite3.Row | None:
    return conn.execute("SELECT * FROM trades WHERE id = ?", (trade_id,)).fetchone()


def resolve_trade(
    conn: sqlite3.Connection,
    trade_id: int,
    outcome: str,
    payout: float,
    profit_loss: float,
    bankroll_after: float,
) -> None:
    conn.execute(
        """UPDATE trades SET resolved_at = ?, outcome = ?, payout = ?,
           profit_loss = ?, bankroll_after = ? WHERE id = ?""",
        (now_iso(), outcome, payout, profit_loss, bankroll_after, trade_id),
    )
    conn.commit()


def count_resolved_trades(conn: sqlite3.Connection) -> int:
    row = conn.execute(
        "SELECT COUNT(*) as cnt FROM trades WHERE outcome != 'PENDING'"
    ).fetchone()
    return row["cnt"]


def get_all_trades(conn: sqlite3.Connection) -> list[sqlite3.Row]:
    return conn.execute("SELECT * FROM trades ORDER BY timestamp DESC").fetchall()


# --- Retros ---

def insert_retro(conn: sqlite3.Connection, retro: dict[str, Any]) -> int:
    cur = conn.execute(
        """INSERT INTO retros
           (trade_id, market_id, resolved_at, data_was_right, sentiment_was_right,
            roc_was_right, news_was_right, structural_was_right, prediction_error,
            lesson, bias_detected, weight_adjustment, retro_type, category)
           VALUES (:trade_id, :market_id, :resolved_at, :data_was_right,
            :sentiment_was_right, :roc_was_right, :news_was_right,
            :structural_was_right, :prediction_error, :lesson, :bias_detected,
            :weight_adjustment, :retro_type, :category)""",
        retro,
    )
    conn.commit()
    return cur.lastrowid


def get_retros_by_type(
    conn: sqlite3.Connection, retro_type: str
) -> list[sqlite3.Row]:
    return conn.execute(
        "SELECT * FROM retros WHERE retro_type = ? ORDER BY resolved_at DESC",
        (retro_type,),
    ).fetchall()


def get_recent_retros(conn: sqlite3.Connection, limit: int = 20) -> list[sqlite3.Row]:
    return conn.execute(
        "SELECT * FROM retros WHERE retro_type = 'normal' ORDER BY resolved_at DESC LIMIT ?",
        (limit,),
    ).fetchall()


def get_retros_for_category(
    conn: sqlite3.Connection, category: str, limit: int = 10
) -> list[sqlite3.Row]:
    return conn.execute(
        "SELECT * FROM retros WHERE category = ? ORDER BY resolved_at DESC LIMIT ?",
        (category, limit),
    ).fetchall()


# --- Denials ---

def insert_denial(conn: sqlite3.Connection, denial: dict[str, Any]) -> int:
    cur = conn.execute(
        """INSERT INTO denials (signal_id, market_id, denied_at)
           VALUES (:signal_id, :market_id, :denied_at)""",
        denial,
    )
    conn.commit()
    return cur.lastrowid


def resolve_denial(
    conn: sqlite3.Connection, denial_id: int, would_have_won: bool, profit: float
) -> None:
    conn.execute(
        "UPDATE denials SET would_have_won = ?, would_have_profit = ? WHERE id = ?",
        (int(would_have_won), profit, denial_id),
    )
    conn.commit()


def get_unresolved_denials(conn: sqlite3.Connection) -> list[sqlite3.Row]:
    return conn.execute(
        "SELECT * FROM denials WHERE would_have_won IS NULL"
    ).fetchall()


def count_denials_for_category(
    conn: sqlite3.Connection, category: str, hours: int = 24
) -> int:
    """Count recent denials for a category (for threshold raising)."""
    row = conn.execute(
        """SELECT COUNT(*) as cnt FROM denials d
           JOIN markets m ON d.market_id = m.id
           WHERE m.category = ?
           AND d.denied_at > datetime('now', ?)""",
        (category, f"-{hours} hours"),
    ).fetchone()
    return row["cnt"]


# --- Bankroll ---

def get_current_bankroll(conn: sqlite3.Connection, starting: float = 10.0) -> float:
    """Calculate current bankroll from trade history."""
    row = conn.execute(
        """SELECT
           COALESCE(
               (SELECT bankroll_after FROM trades
                WHERE bankroll_after IS NOT NULL
                ORDER BY timestamp DESC LIMIT 1),
               ?
           ) as bankroll""",
        (starting,),
    ).fetchone()
    return row["bankroll"]


def get_portfolio_exposure(conn: sqlite3.Connection) -> float:
    """Sum of capital committed to open positions."""
    row = conn.execute(
        """SELECT COALESCE(SUM(contracts * price_paid), 0) / 100.0 as exposure
           FROM trades WHERE outcome = 'PENDING'"""
    ).fetchone()
    return row["exposure"]
```

```python
# cipher/kalshi.py
"""Kalshi REST API client with RSA key pair authentication."""
from __future__ import annotations

import base64
import hashlib
import time
from datetime import datetime, timezone, timedelta
from pathlib import Path
from typing import Any

import httpx
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.asymmetric import padding, utils

from cipher.config import Config


class KalshiClient:
    """Synchronous Kalshi API client using RSA-signed requests."""

    def __init__(self, config: Config) -> None:
        self.config = config
        self.base_url = config.kalshi_base_url
        self.api_key_id = config.kalshi_api_key_id
        self._private_key = self._load_private_key(config.kalshi_rsa_key_path)
        self._client = httpx.Client(timeout=30.0)

    def _load_private_key(self, key_path: Path):
        """Load RSA private key from PEM file."""
        with open(key_path, "rb") as f:
            return serialization.load_pem_private_key(f.read(), password=None)

    def _sign_request(self, method: str, path: str, timestamp_ms: int) -> str:
        """
        Kalshi RSA auth: sign the string 'timestamp_ms + method + path'
        with the private key using PSS padding.
        """
        message = f"{timestamp_ms}{method}{path}".encode("utf-8")
        # Kalshi uses RSA-PSS with SHA-256
        signature = self._private_key.sign(
            message,
            padding.PSS(
                mgf=padding.MGF1(hashes.SHA256()),
                salt_length=padding.PSS.MAX_LENGTH,
            ),
            hashes.SHA256(),
        )
        return base64.b64encode(signature).decode("utf-8")

    def _headers(self, method: str, path: str) -> dict[str, str]:
        """Build auth headers for a request."""
        timestamp_ms = int(time.time() * 1000)
        signature = self._sign_request(method, path, timestamp_ms)
        return {
            "KALSHI-ACCESS-KEY": self.api_key_id,
            "KALSHI-ACCESS-SIGNATURE": signature,
            "KALSHI-ACCESS-TIMESTAMP": str(timestamp_ms),
            "Content-Type": "application/json",
            "Accept": "application/json",
        }

    def _request(
        self, method: str, path: str, params: dict | None = None, json: dict | None = None
    ) -> dict:
        """Make an authenticated request."""
        url = f"{self.base_url}{path}"
        headers = self._headers(method.upper(), path)
        resp = self._client.request(method, url, headers=headers, params=params, json=json)
        resp.raise_for_status()
        return resp.json()

    def get_balance(self) -> float:
        """Get account balance in dollars."""
        data = self._request("GET", "/portfolio/balance")
        # Balance is returned in cents
        return data["balance"] / 100.0

    def get_active_markets(
        self,
        limit: int = 100,
        status: str = "open",
        max_close_hours: int = 24,
    ) -> list[dict[str, Any]]:
        """
        Fetch active markets closing within max_close_hours.
        Returns list of market dicts normalized to our schema.
        """
        # Kalshi uses cursor-based pagination
        cursor: str | None = None
        all_markets: list[dict] = []
        cutoff = datetime.now(timezone.utc) + timedelta(hours=max_close_hours)

        while True:
            params: dict[str, Any] = {
                "limit": limit,
                "status": status,
            }
            if cursor:
                params["cursor"] = cursor

            data = self._request("GET", "/markets", params=params)
            markets = data.get("markets", [])

            for m in markets:
                close_time = m.get("close_time") or m.get("expected_expiration_time", "")
                if close_time:
                    try:
                        ct = datetime.fromisoformat(close_time.replace("Z", "+00:00"))
                        if ct > cutoff:
                            continue
                    except ValueError:
                        continue

                all_markets.append(self._normalize_market(m))

            cursor = data.get("cursor")
            if not cursor or not markets:
                break

        return all_markets

    def _normalize_market(self, m: dict) -> dict[str, Any]:
        """Normalize Kalshi API response to our market schema."""
        return {
            "id": m["event_ticker"] if "event_ticker" in m else m.get("ticker", ""),
            "ticker": m.get("ticker", ""),
            "title": m.get("title", ""),
            "category": m.get("category", ""),
            "close_time": m.get("close_time") or m.get("expected_expiration_time", ""),
            "status": "active",
            "last_price": m.get("last_price"),
            "volume": m.get("volume", 0),
            "yes_bid": m.get("yes_bid"),
            "yes_ask": m.get("yes_ask"),
            "no_bid": m.get("no_bid"),
            "no_ask": m.get("no_ask"),
        }

    def get_candles(
        self,
        ticker: str,
        period_interval: int = 1,
        series_type: str = "price",
    ) -> list[dict[str, Any]]:
        """
        Get candlestick data for a market.
        period_interval: minutes (1, 5, 15, 60, etc.)
        Returns list of candle dicts normalized to our schema.
        """
        params = {
            "series_ticker": ticker,
            "period_interval": period_interval,
        }
        data = self._request("GET", f"/series/{ticker}/markets/{ticker}/candlesticks", params=params)
        candles = data.get("candlesticks", [])
        return [
            {
                "market_id": ticker,
                "timestamp": c["end_period_ts"],
                "open": c.get("price", {}).get("open", 0),
                "high": c.get("price", {}).get("high", 0),
                "low": c.get("price", {}).get("low", 0),
                "close": c.get("price", {}).get("close", 0),
                "volume": c.get("volume", 0),
            }
            for c in candles
        ]

    def get_orderbook(self, ticker: str) -> dict[str, Any]:
        """Get the order book for a market."""
        return self._request("GET", f"/markets/{ticker}/orderbook")

    def place_order(
        self,
        ticker: str,
        side: str,
        contracts: int,
        price: int,
        order_type: str = "limit",
    ) -> dict[str, Any]:
        """
        Place an order on Kalshi.
        side: 'yes' or 'no'
        price: in cents (1-99)
        Returns order confirmation dict.
        """
        payload = {
            "ticker": ticker,
            "action": "buy",
            "side": side,
            "count": contracts,
            "type": order_type,
            "yes_price" if side == "yes" else "no_price": price,
        }
        return self._request("POST", "/portfolio/orders", json=payload)

    def get_positions(self) -> list[dict[str, Any]]:
        """Get current portfolio positions."""
        data = self._request("GET", "/portfolio/positions")
        return data.get("market_positions", [])

    def get_market_by_ticker(self, ticker: str) -> dict[str, Any]:
        """Get a single market by ticker."""
        data = self._request("GET", f"/markets/{ticker}")
        return self._normalize_market(data.get("market", data))

    def close(self) -> None:
        self._client.close()
```

```python
# cipher/candles.py
"""Time series feature extraction from candlestick data."""
from __future__ import annotations

from dataclasses import dataclass

import numpy as np
from numpy.typing import NDArray


@dataclass
class CandleFeatures:
    """Features extracted from a price history."""
    momentum: float           # linear regression slope, positive = bullish
    volume_spike: float       # current volume vs rolling average (ratio)
    volatility: float         # std of returns
    mean_reversion: float     # distance from rolling mean (signed)
    bid_ask_spread: int       # current spread in cents
    time_decay: float         # hours remaining until close, normalized 0-1
    n_candles: int            # how many candles we computed from


def extract_features(
    closes: NDArray[np.float64],
    volumes: NDArray[np.float64],
    yes_bid: int,
    yes_ask: int,
    hours_remaining: float,
    max_hours: float = 24.0,
) -> CandleFeatures:
    """
    Extract trading features from candle data.

    closes: array of close prices (cents), oldest first
    volumes: array of volumes, oldest first
    yes_bid, yes_ask: current bid/ask
    hours_remaining: hours until market close
    max_hours: max hours for time decay normalization
    """
    n = len(closes)
    if n < 2:
        return CandleFeatures(
            momentum=0.0,
            volume_spike=0.0,
            volatility=0.0,
            mean_reversion=0.0,
            bid_ask_spread=yes_ask - yes_bid if yes_ask and yes_bid else 0,
            time_decay=min(hours_remaining / max_hours, 1.0),
            n_candles=n,
        )

    # Momentum: linear regression slope of close prices
    x = np.arange(n, dtype=np.float64)
    slope, _ = np.polyfit(x, closes, 1)
    # Normalize slope by price level to get percentage momentum
    avg_price = np.mean(closes)
    momentum = slope / avg_price if avg_price > 0 else 0.0

    # Volume spike: latest volume vs rolling average (20-period or available)
    window = min(20, n)
    rolling_vol_avg = np.mean(volumes[-window:])
    current_vol = volumes[-1]
    volume_spike = (
        current_vol / rolling_vol_avg if rolling_vol_avg > 0 else 0.0
    )

    # Volatility: standard deviation of returns
    returns = np.diff(closes) / closes[:-1]
    returns = returns[np.isfinite(returns)]
    volatility = float(np.std(returns)) if len(returns) > 0 else 0.0

    # Mean reversion: distance from 20-period rolling mean
    rolling_mean = np.mean(closes[-window:])
    current_price = closes[-1]
    mean_reversion = (
        (current_price - rolling_mean) / rolling_mean if rolling_mean > 0 else 0.0
    )

    # Bid-ask spread
    spread = (yes_ask - yes_bid) if (yes_ask is not None and yes_bid is not None) else 0

    # Time decay: normalized 0-1 (1 = just opened, 0 = about to close)
    time_decay = max(0.0, min(hours_remaining / max_hours, 1.0))

    return CandleFeatures(
        momentum=float(momentum),
        volume_spike=float(volume_spike),
        volatility=float(volatility),
        mean_reversion=float(mean_reversion),
        bid_ask_spread=int(spread),
        time_decay=float(time_decay),
        n_candles=n,
    )
```

```python
# cipher/signals/rate_of_change.py
"""
Signal 3: Rate of Change.

Price velocity, acceleration, volume-weighted direction,
unjustified momentum detection.

Returns score in [-1, 1] and confidence in [0, 1].
Positive score = bullish momentum, negative = bearish.
"""
from __future__ import annotations

from dataclasses import dataclass

import numpy as np
from numpy.typing import NDArray


@dataclass
class RateOfChangeResult:
    score: float       # -1 to 1, directional
    confidence: float  # 0 to 1
    velocity: float    # price change per period
    acceleration: float  # change in velocity
    volume_direction: float  # volume-weighted price direction
    unjustified: bool  # momentum without volume support


def compute_rate_of_change(
    closes: NDArray[np.float64],
    volumes: NDArray[np.float64],
    min_periods: int = 5,
) -> RateOfChangeResult:
    """
    Compute rate-of-change signal from price and volume data.

    closes: array of close prices, oldest first
    volumes: array of volumes, oldest first
    """
    n = len(closes)
    if n < min_periods:
        return RateOfChangeResult(
            score=0.0, confidence=0.0,
            velocity=0.0, acceleration=0.0,
            volume_direction=0.0, unjustified=False,
        )

    # Price velocity: average of last 5 returns
    returns = np.diff(closes) / closes[:-1]
    returns = returns[np.isfinite(returns)]
    if len(returns) < 2:
        return RateOfChangeResult(
            score=0.0, confidence=0.0,
            velocity=0.0, acceleration=0.0,
            volume_direction=0.0, unjustified=False,
        )

    velocity = float(np.mean(returns[-min_periods:]))

    # Price acceleration: change in velocity (second derivative)
    if len(returns) >= min_periods + 1:
        recent = returns[-min_periods:]
        earlier = returns[-(min_periods + 1):-1]
        acceleration = float(np.mean(recent) - np.mean(earlier))
    else:
        acceleration = 0.0

    # Volume-weighted direction
    # Positive when volume is high during price increases
    vol_weights = volumes[-len(returns):]
    if len(vol_weights) == len(returns) and np.sum(vol_weights) > 0:
        volume_direction = float(
            np.sum(returns * vol_weights) / np.sum(vol_weights)
        )
    else:
        volume_direction = velocity

    # Unjustified momentum: strong price move with declining or flat volume
    recent_vol = volumes[-min_periods:] if len(volumes) >= min_periods else volumes
    vol_trend = 0.0
    if len(recent_vol) >= 2:
        vx = np.arange(len(recent_vol), dtype=np.float64)
        vol_slope, _ = np.polyfit(vx, recent_vol, 1)
        avg_vol = np.mean(recent_vol)
        vol_trend = vol_slope / avg_vol if avg_vol > 0 else 0.0

    abs_velocity = abs(velocity)
    unjustified = abs_velocity > 0.02 and vol_trend < 0.0

    # Composite score: blend velocity, acceleration, volume direction
    # Clamp to [-1, 1]
    raw_score = (velocity * 50) + (acceleration * 30) + (volume_direction * 20)
    score = float(np.clip(raw_score, -1.0, 1.0))

    # Confidence: higher when signals agree on direction and magnitude is strong
    signals_agree = np.sign(velocity) == np.sign(volume_direction)
    magnitude = min(abs(score), 1.0)
    confidence = magnitude * (0.8 if signals_agree else 0.5)
    if unjustified:
        confidence *= 0.6  # reduce confidence for unjustified moves

    return RateOfChangeResult(
        score=score,
        confidence=float(np.clip(confidence, 0.0, 1.0)),
        velocity=velocity,
        acceleration=acceleration,
        volume_direction=volume_direction,
        unjustified=unjustified,
    )
```

```python
# cipher/signals/structural.py
"""
Signal 5: Structural Mispricing.

Mechanical inefficiencies unrelated to information:
- Time decay distortion (near-expiry not converged)
- Liquidity gaps (thin order books)
- Stale markets (no recent price updates)

Returns score in [-1, 1] and confidence in [0, 1].
Positive = structural edge for YES, negative = edge for NO.
"""
from __future__ import annotations

from dataclasses import dataclass


@dataclass
class StructuralResult:
    score: float       # -1 to 1
    confidence: float  # 0 to 1
    time_decay_distortion: float  # how far from convergence
    liquidity_score: float        # 0 (thin) to 1 (thick)
    is_stale: bool                # no recent volume


def compute_structural(
    last_price: int,
    yes_bid: int | None,
    yes_ask: int | None,
    volume: int,
    hours_remaining: float,
    recent_volume: int = 0,
    stale_threshold_minutes: int = 30,
    minutes_since_last_trade: float = 0.0,
) -> StructuralResult:
    """
    Compute structural mispricing signal.

    last_price: in cents (0-100)
    yes_bid, yes_ask: current bid/ask in cents
    volume: total market volume
    hours_remaining: until close
    recent_volume: volume in last N candles
    minutes_since_last_trade: for staleness detection
    """
    bid = yes_bid or 0
    ask = yes_ask or 100

    # --- Time decay distortion ---
    # Near expiry, prices should converge toward 0 or 100.
    # If they're stuck in the middle, there's a structural opportunity.
    if hours_remaining <= 0:
        time_decay_distortion = 0.0
    elif hours_remaining < 2:
        # Within 2 hours of close, price should be moving toward extremes
        distance_from_extreme = min(last_price, 100 - last_price)
        # Normalize: 50 is max distance, scale by time urgency
        urgency = 1.0 - (hours_remaining / 2.0)
        time_decay_distortion = (distance_from_extreme / 50.0) * urgency
    else:
        time_decay_distortion = 0.0

    # --- Liquidity score ---
    spread = ask - bid
    if spread <= 0:
        spread = 1
    # Tight spread = good liquidity. Wide spread = structural opportunity
    # Spread of 1-2 cents = liquid. Spread of 10+ = illiquid
    liquidity_score = max(0.0, 1.0 - (spread - 1) / 15.0)

    # --- Staleness ---
    is_stale = (
        minutes_since_last_trade > stale_threshold_minutes
        or (volume > 100 and recent_volume == 0)
    )

    # --- Composite score ---
    # Direction: if price is stuck in the middle near expiry,
    # lean toward the closer extreme
    if last_price > 50:
        direction = 1.0   # YES is more likely
    elif last_price < 50:
        direction = -1.0  # NO is more likely
    else:
        direction = 0.0   # no directional edge

    # Score components
    score_components = []
    weights = []

    if time_decay_distortion > 0.1:
        score_components.append(direction * time_decay_distortion)
        weights.append(0.5)

    if liquidity_score < 0.5:
        # Wide spread means favorable entry is possible
        score_components.append(direction * (1.0 - liquidity_score) * 0.5)
        weights.append(0.3)

    if is_stale:
        # Stale market may not reflect current info — directional lean
        score_components.append(direction * 0.3)
        weights.append(0.2)

    if not score_components:
        return StructuralResult(
            score=0.0, confidence=0.0,
            time_decay_distortion=time_decay_distortion,
            liquidity_score=liquidity_score,
            is_stale=is_stale,
        )

    total_weight = sum(weights)
    score = sum(s * w for s, w in zip(score_components, weights)) / total_weight
    score = max(-1.0, min(1.0, score))

    # Confidence scales with how many structural factors are present
    factors_present = sum([
        time_decay_distortion > 0.1,
        liquidity_score < 0.5,
        is_stale,
    ])
    confidence = min(0.3 * factors_present + abs(score) * 0.3, 1.0)

    return StructuralResult(
        score=score,
        confidence=confidence,
        time_decay_distortion=time_decay_distortion,
        liquidity_score=liquidity_score,
        is_stale=is_stale,
    )
```

```python
"""
Signal 1: Data Disparity.

V1: LLM-driven. Asks Kimi K2.6 to estimate true probability from public knowledge.
V1.1: Will add FRED API, NWS API, polling aggregates.

The gap between market price and data-implied probability is the edge.
"""
from __future__ import annotations

from dataclasses import dataclass


@dataclass
class DataDisparityResult:
    score: float       # -1 to 1, directional (positive = YES underpriced)
    confidence: float  # 0 to 1
    market_probability: float   # what the market says (from price)
    estimated_probability: float  # what data says


def compute_data_disparity(
    market_price: int,
    llm_estimated_probability: float,
    llm_confidence: float,
) -> DataDisparityResult:
    """
    Compute data disparity from market price and LLM estimate.

    market_price: in cents (0-100), directly maps to implied probability
    llm_estimated_probability: 0-100, from Kimi K2.6 analysis
    llm_confidence: 0-1, from Kimi K2.6
    """
    market_prob = market_price / 100.0
    est_prob = llm_estimated_probability / 100.0

    # Gap: positive means market underprices YES
    gap = est_prob - market_prob

    # Score: scale the gap. A 10-point gap is meaningful.
    # Clamp to [-1, 1]
    score = max(-1.0, min(1.0, gap * 5.0))

    # Confidence: blend LLM confidence with gap magnitude
    # Larger gaps with high LLM confidence = higher signal confidence
    confidence = llm_confidence * min(abs(gap) * 10.0, 1.0)

    return DataDisparityResult(
        score=score,
        confidence=confidence,
        market_probability=market_prob,
        estimated_probability=est_prob,
    )
```

```python
    Compute sentiment signal from LLM analysis.

    llm_sentiment_confidence: 0-1 from Kimi K2.6
    bias_flags: list of detected biases from Kimi K2.6

    confidence = llm_sentiment_confidence * bias_multiplier

    return SentimentResult(
        score=score,
        confidence=confidence,
        biases=bias_flags,
    )
```

```python
# cipher/sizing.py
"""
Kelly criterion position sizing for bounded prediction markets.

Based on arXiv 2412.14144: Application of the Kelly Criterion to Prediction Markets.

f* = (p * b - q) / b

where:
  p = estimated true probability
  b = payout odds (net gain per dollar risked)
  q = 1 - p

For prediction markets:
  - Buy YES at price c (cents): risk c, win (100 - c)
  - Buy NO at price (100 - c): risk (100 - c), win c
  - b = (100 - c) / c for YES, or c / (100 - c) for NO
"""
from __future__ import annotations

from dataclasses import dataclass


@dataclass
class PositionSize:
    contracts: int
    price_per_contract: int   # cents
    total_cost: float         # dollars
    max_payout: float         # dollars (if win)
    expected_profit: float    # dollars (EV)
    kelly_fraction: float     # raw Kelly f*
    capped_fraction: float    # after caps
    bankroll_pct: float       # percentage of bankroll


def compute_kelly(
    estimated_prob: float,
    market_price: int,
    direction: str,
    bankroll: float,
    max_position_pct: float = 0.20,
    min_cost: float = 0.01,
) -> PositionSize:
    """
    Compute Kelly-optimal position size.

    estimated_prob: 0.0 to 1.0, our estimate of true probability of YES
    market_price: current YES price in cents (1-99)
    direction: 'YES' or 'NO'
    bankroll: current bankroll in dollars
    max_position_pct: max fraction of bankroll per trade
    min_cost: minimum trade size in dollars
    """
    if direction == "YES":
        cost_cents = market_price
        p = estimated_prob
    else:
        cost_cents = 100 - market_price
        p = 1.0 - estimated_prob

    q = 1.0 - p
    win_cents = 100 - cost_cents
    cost_dollars = cost_cents / 100.0

    # Payout odds: net gain per dollar risked
    if cost_cents <= 0 or cost_cents >= 100:
        return _zero_position(market_price, direction)

    b = win_cents / cost_cents

    # Kelly fraction
    kelly_f = (p * b - q) / b if b > 0 else 0.0

    # Don't bet if Kelly is negative (no edge)
    if kelly_f <= 0:
        return _zero_position(market_price, direction)

    # Cap at max_position_pct
    capped_f = min(kelly_f, max_position_pct)

    # Dollar amount to risk
    risk_dollars = bankroll * capped_f

    # Number of contracts (whole number, at least 1 if any edge)
    contracts = max(1, int(risk_dollars / cost_dollars)) if risk_dollars >= min_cost else 0

    if contracts == 0:
        return _zero_position(market_price, direction)

    total_cost = contracts * cost_dollars
    max_payout = contracts * 1.00  # each contract pays $1 on win
    expected_profit = p * (max_payout - total_cost) - q * total_cost

    return PositionSize(
        contracts=contracts,
        price_per_contract=cost_cents,
        total_cost=round(total_cost, 4),
        max_payout=round(max_payout, 2),
        expected_profit=round(expected_profit, 4),
        kelly_fraction=round(kelly_f, 6),
        capped_fraction=round(capped_f, 6),
        bankroll_pct=round(total_cost / bankroll, 4) if bankroll > 0 else 0.0,
    )


def _zero_position(market_price: int, direction: str) -> PositionSize:
    cost = market_price if direction == "YES" else 100 - market_price
    return PositionSize(
        contracts=0,
        price_per_contract=cost,
        total_cost=0.0,
        max_payout=0.0,
        expected_profit=0.0,
        kelly_fraction=0.0,
        capped_fraction=0.0,
        bankroll_pct=0.0,
    )


def compute_ev(
    estimated_prob: float,
    market_price: int,
    direction: str,
    contracts: int,
    fees: float = 0.0,
    inference_cost: float = 0.10,
) -> float:
    """
    Compute expected value of a trade.

    Returns EV in dollars. Must exceed fees + inference cost to pass entry rules.
    """
    if direction == "YES":
        cost_cents = market_price
        p = estimated_prob
    else:
        cost_cents = 100 - market_price
        p = 1.0 - estimated_prob

    cost_per = cost_cents / 100.0
    total_cost = contracts * cost_per
    total_payout = contracts * 1.00

    ev = p * (total_payout - total_cost) - (1.0 - p) * total_cost
    return round(ev - fees - inference_cost, 4)
```

Create `cipher/engine.py`:

```python
# cipher/engine.py
"""
Decision engine.

Combines 5 signal scores with weights, applies entry rules,
determines whether to pitch a trade.

Deterministic: same inputs always produce the same output.
"""
from __future__ import annotations

from dataclasses import dataclass
from typing import Any

from cipher.config import Config, SignalWeights
from cipher.sizing import PositionSize, compute_ev, compute_kelly


@dataclass
class SignalBundle:
    """All 5 signal scores and confidences."""
    data_disparity: float         # -1 to 1
    data_disparity_conf: float    # 0 to 1
    sentiment_bias: float         # -1 to 1
    sentiment_bias_conf: float    # 0 to 1
    rate_of_change: float         # -1 to 1
    rate_of_change_conf: float    # 0 to 1
    news_timing: float            # -1 to 1
    news_timing_conf: float       # 0 to 1
    structural: float             # -1 to 1
    structural_conf: float        # 0 to 1


@dataclass
class Decision:
    """Output of the decision engine."""
    action: str                   # PITCH, SKIP
    direction: str | None         # YES, NO, or None if SKIP
    combined_score: float
    confidence: float
    position: PositionSize | None
    rejection_reasons: list[str]  # why we skipped (empty if pitching)
    ev: float                     # expected value in dollars
    reasoning: str


def evaluate(
    signals: SignalBundle,
    market_price: int,
    estimated_prob: float,
    bankroll: float,
    portfolio_exposure: float,
    open_positions: int,
    category: str,
    config: Config,
    reasoning: str = "",
    category_denial_count: int = 0,
) -> Decision:
    """
    Apply entry rules and determine whether to pitch.

    This function is pure and deterministic.
    """
    weights = config.weights
    rejection_reasons: list[str] = []

    # --- Compute combined score ---
    # Use absolute values weighted by confidence for magnitude,
    # then determine direction from sign agreement.
    available_signals = []
    weighted_scores = []

    signal_list = [
        ("data_disparity", signals.data_disparity, signals.data_disparity_conf, weights.data_disparity),
        ("sentiment_bias", signals.sentiment_bias, signals.sentiment_bias_conf, weights.sentiment_bias),
        ("rate_of_change", signals.rate_of_change, signals.rate_of_change_conf, weights.rate_of_change),
        ("news_timing", signals.news_timing, signals.news_timing_conf, weights.news_timing),
        ("structural", signals.structural, signals.structural_conf, weights.structural),
    ]

    for name, score, conf, weight in signal_list:
        if conf > 0:  # signal is available
            available_signals.append((name, score, conf, weight))
            weighted_scores.append(score * conf * weight)

    if not available_signals:
        return Decision(
            action="SKIP", direction=None,
            combined_score=0.0, confidence=0.0,
            position=None, rejection_reasons=["No available signals"],
            ev=0.0, reasoning=reasoning,
        )

    # Combined score: weighted sum (preserves sign/direction)
    total_weight = sum(w for _, _, _, w in available_signals)
    combined_score = sum(weighted_scores) / total_weight if total_weight > 0 else 0.0

    # Confidence: magnitude of combined score, scaled by average signal confidence
    avg_conf = sum(c for _, _, c, _ in available_signals) / len(available_signals)
    confidence = abs(combined_score) * avg_conf

    # Direction
    if combined_score > 0:
        direction = "YES"
    elif combined_score < 0:
        direction = "NO"
    else:
        direction = None

    # --- Entry rule 1: Confidence > threshold ---
    min_conf = config.min_confidence
    # Raise threshold if 3+ recent denials in this category
    if category_denial_count >= 3:
        min_conf += 0.1

    if confidence <= min_conf:
        rejection_reasons.append(
            f"Confidence {confidence:.2f} below threshold {min_conf:.2f}"
        )

    # --- Entry rule 2: At least 2 signals agree on direction ---
    if direction:
        agreeing = sum(
            1 for _, score, conf, _ in available_signals
            if conf > 0 and (
                (direction == "YES" and score > 0) or
                (direction == "NO" and score < 0)
            )
        )
        if agreeing < config.min_signals_agree:
            rejection_reasons.append(
                f"Only {agreeing} signals agree (need {config.min_signals_agree})"
            )

    # --- Entry rule 3: EV check (computed after sizing) ---
    # Deferred until we have position size

    # --- Entry rule 4: Portfolio exposure < 60% ---
    if portfolio_exposure / bankroll > config.max_portfolio_exposure_pct if bankroll > 0 else True:
        rejection_reasons.append(
            f"Portfolio exposure {portfolio_exposure:.2f} exceeds "
            f"{config.max_portfolio_exposure_pct * 100:.0f}% of bankroll"
        )

    # --- Entry rule 5: No more than 3 open positions ---
    if open_positions >= config.max_open_positions:
        rejection_reasons.append(
            f"Already {open_positions} open positions (max {config.max_open_positions})"
        )

    # --- Entry rule 6: No contradicting signal > 0.7 ---
    if direction:
        for name, score, conf, _ in available_signals:
            contradicts = (
                (direction == "YES" and score < 0) or
                (direction == "NO" and score > 0)
            )
            if contradicts and abs(score) > config.max_contradicting_signal:
                rejection_reasons.append(
                    f"Signal '{name}' contradicts at {score:.2f} (threshold {config.max_contradicting_signal})"
                )

    # --- Position sizing ---
    position = None
    ev = 0.0
    if direction and not rejection_reasons:
        position = compute_kelly(
            estimated_prob=estimated_prob / 100.0 if estimated_prob > 1 else estimated_prob,
            market_price=market_price,
            direction=direction,
            bankroll=bankroll,
            max_position_pct=config.max_position_pct,
        )
        if position.contracts == 0:
            rejection_reasons.append("Kelly criterion says no bet (no edge or bankroll too low)")
        else:
            # Entry rule 3: EV check
            ev = compute_ev(
                estimated_prob=estimated_prob / 100.0 if estimated_prob > 1 else estimated_prob,
                market_price=market_price,
                direction=direction,
                contracts=position.contracts,
            )
            if ev <= 0:
                rejection_reasons.append(f"Negative EV: ${ev:.4f}")

    action = "PITCH" if not rejection_reasons and position and position.contracts > 0 else "SKIP"

    return Decision(
        action=action,
        direction=direction,
        combined_score=round(combined_score, 4),
        confidence=round(confidence, 4),
        position=position,
        rejection_reasons=rejection_reasons,
        ev=round(ev, 4),
        reasoning=reasoning,
    )
```

```python
from __future__ import annotations

def format_pitch(
    decision: Decision,
    market_title: str,
    market_price: int,
    estimated_prob: float,
    bankroll: float,
    signal_scores: dict[str, float],
) -> str:
    """Format a pitch message for ntfy delivery."""
    pos = decision.position
    if not pos:
        return ""

    edge = abs(estimated_prob - market_price)
    exposure_pct = (pos.total_cost / bankroll * 100) if bankroll > 0 else 0

    lines = [
        f"CIPHER PITCH -- {decision.confidence:.2f} confidence",
        "",
        f'Market: "{market_title}"',
        f"Direction: {decision.direction}",
        f"Current price: {market_price}c (market says {market_price}%)",
        f"My estimate: {estimated_prob:.0f}%",
        f"Edge: {edge:.0f} points",
        "",
        "Signals:",
        f"  Data disparity: {signal_scores.get('data_disparity', 'N/A'):.2f}"
            if isinstance(signal_scores.get('data_disparity'), (int, float))
            else f"  Data disparity: N/A",
        f"  Sentiment bias: {signal_scores.get('sentiment_bias', 'N/A'):.2f}"
            if isinstance(signal_scores.get('sentiment_bias'), (int, float))
            else f"  Sentiment bias: N/A",
        f"  Rate of change: {signal_scores.get('rate_of_change', 'N/A'):.2f}"
            if isinstance(signal_scores.get('rate_of_change'), (int, float))
            else f"  Rate of change: N/A",
        f"  News timing: {signal_scores.get('news_timing', 'N/A'):.2f}"
            if isinstance(signal_scores.get('news_timing'), (int, float))
            else f"  News timing: N/A",
        f"  Structural: {signal_scores.get('structural', 'N/A'):.2f}"
            if isinstance(signal_scores.get('structural'), (int, float))
            else f"  Structural: N/A",
        "",
        f"Position: {pos.contracts} contracts at {pos.price_per_contract}c (${pos.total_cost:.2f} risk)",
        f"Max payout: ${pos.max_payout:.2f} -> profit ${pos.max_payout - pos.total_cost:.2f}",
        f"Bankroll: ${bankroll:.2f} -> exposure {exposure_pct:.1f}%",
        f"EV: ${decision.ev:.4f}",
    ]

    return "\n".join(lines)


def send_pitch(
    topic: str,
    message: str,
    signal_id: int,
) -> bool:
    """
    Send pitch via ntfy.sh with approve/deny action buttons.

    Returns True if sent successfully.
    """
    try:
        resp = httpx.post(
            f"{NTFY_URL}/{topic}",
            data=message.encode("utf-8"),
            headers={
                "Title": "Cipher Trade Pitch",
                "Priority": "high",
                "Tags": "chart_with_upwards_trend,dollar",
                "Actions": (
                    f"http, Approve, https://ntfy.sh/{topic}/publish, "
                    f'body="approve:{signal_id}", method=POST; '
                    f"http, Deny, https://ntfy.sh/{topic}/publish, "
                    f'body="deny:{signal_id}", method=POST'
                ),
            },
        )
        return resp.status_code == 200
    except httpx.HTTPError:
        return False


def send_notification(topic: str, title: str, message: str, priority: str = "default") -> bool:
    """Send a generic notification via ntfy."""
    try:
        resp = httpx.post(
            f"{NTFY_URL}/{topic}",
            data=message.encode("utf-8"),
            headers={
                "Title": title,
                "Priority": priority,
            },
        )
        return resp.status_code == 200
    except httpx.HTTPError:
        return False


def check_pitch_expiry(conn: sqlite3.Connection, expiry_minutes: int = 30) -> bool:
    """
    Check if the pending pitch has expired.
    Returns True if a pitch was expired (and updated).
    """
    pitch = get_pending_pitch(conn)
    if not pitch:
        return False

    pitched_at = datetime.fromisoformat(pitch["timestamp"])
    now = datetime.now(timezone.utc)
    if now - pitched_at > timedelta(minutes=expiry_minutes):
        update_signal_action(conn, pitch["id"], "EXPIRED")
        return True
    return False


def handle_approve(conn: sqlite3.Connection) -> int | None:
    """
    Approve the pending pitch.
    Returns the signal_id if approved, None if no pending pitch.
    """
    pitch = get_pending_pitch(conn)
    if not pitch:
        return None
    update_signal_action(conn, pitch["id"], "APPROVED")
    return pitch["id"]


def handle_deny(conn: sqlite3.Connection) -> int | None:
    """
    Deny the pending pitch and log the denial.
    Returns the signal_id if denied, None if no pending pitch.
    """
    pitch = get_pending_pitch(conn)
    if not pitch:
        return None

    update_signal_action(conn, pitch["id"], "DENIED")
    insert_denial(conn, {
        "signal_id": pitch["id"],
        "market_id": pitch["market_id"],
        "denied_at": now_iso(),
    })
    return pitch["id"]
```

```python
def execute_trade(
    conn: sqlite3.Connection,
    kalshi: KalshiClient,
    signal_id: int,
    config: Config,
) -> int | None:
    """
    Execute an approved trade on Kalshi.

    Returns trade_id if successful, None if failed.
    """
    signal = get_signal(conn, signal_id)
    if not signal:
        return None

    bankroll = get_current_bankroll(conn, config.starting_bankroll)
    direction = signal["direction"]
    market_id = signal["market_id"]

    # Recompute position from signal data (or use stored values)
    from cipher.sizing import compute_kelly

    estimated_prob = signal["confidence"]
    market_price = 50  # will be overridden by live data

    # Get fresh market data
    try:
        market_data = kalshi.get_market_by_ticker(market_id)
        if direction == "YES":
            market_price = market_data.get("yes_ask") or market_data.get("last_price") or 50
        else:
            market_price = market_data.get("no_ask") or (100 - (market_data.get("last_price") or 50))
    except Exception:
        # Fall back to signal-time data
        pass

    position = compute_kelly(
        estimated_prob=estimated_prob,
        market_price=market_price,
        direction=direction,
        bankroll=bankroll,
        max_position_pct=config.max_position_pct,
    )

    if position.contracts == 0:
        return None

    # Place order on Kalshi
    try:
        order = kalshi.place_order(
            ticker=market_id,
            side=direction.lower(),
            contracts=position.contracts,
            price=position.price_per_contract,
        )
    except Exception as e:
        send_notification(
            config.ntfy_topic,
            "Cipher: Order Failed",
            f"Failed to place order for {market_id}: {e}",
            priority="high",
        )
        return None

    # Calculate fees (Kalshi charges ~$0.01 per contract per side)
    fees = position.contracts * 0.01

    trade_id = insert_trade(conn, {
        "signal_id": signal_id,
        "market_id": market_id,
        "timestamp": now_iso(),
        "direction": direction,
        "contracts": position.contracts,
        "price_paid": position.price_per_contract,
        "fees": fees,
        "outcome": "PENDING",
        "bankroll_before": bankroll,
    })

    send_notification(
        config.ntfy_topic,
        "Cipher: Trade Executed",
        f"Bought {position.contracts}x {direction} on {market_id} "
        f"at {position.price_per_contract}c (${position.total_cost:.2f})",
    )

    return trade_id


def check_resolutions(
    conn: sqlite3.Connection,
    kalshi: KalshiClient,
    config: Config,
) -> list[int]:
    """
    Check all open trades for resolution.
    Returns list of trade_ids that resolved.
    """
    open_trades = get_open_trades(conn)
    resolved = []

    for trade in open_trades:
        try:
            market = kalshi.get_market_by_ticker(trade["market_id"])
        except Exception:
            continue

        status = market.get("status", "active")
        if status not in ("resolved", "closed"):
            continue

        # Determine outcome
        result = market.get("result", "")
        direction = trade["direction"]

        if result == "yes":
            won = direction == "YES"
        elif result == "no":
            won = direction == "NO"
        else:
            continue  # unknown result

        outcome = "WIN" if won else "LOSS"
        contracts = trade["contracts"]
        price_paid = trade["price_paid"]
        cost = contracts * price_paid / 100.0
        fees = trade["fees"] or 0.0

        if won:
            payout = contracts * 1.00  # $1 per contract
            profit_loss = payout - cost - fees
        else:
            payout = 0.0
            profit_loss = -(cost + fees)

        bankroll = get_current_bankroll(conn, config.starting_bankroll)
        new_bankroll = bankroll + profit_loss

        resolve_trade(conn, trade["id"], outcome, payout, profit_loss, new_bankroll)
        resolved.append(trade["id"])

        send_notification(
            config.ntfy_topic,
            f"Cipher: {outcome}",
            f"{direction} on {trade['market_id']}: "
            f"{'$' + f'{profit_loss:.2f}' if profit_loss >= 0 else '-$' + f'{abs(profit_loss):.2f}'}"
            f" | Bankroll: ${new_bankroll:.2f}",
            priority="high" if outcome == "LOSS" else "default",
        )

    return resolved


def should_skim(bankroll: float, config: Config) -> float:
    """
    Calculate skim amount if bankroll exceeds threshold.
    Returns dollar amount to skim (0 if below threshold).
    """
    if bankroll <= config.skim_threshold:
        return 0.0
    profit_above = bankroll - config.skim_threshold
    return round(profit_above * config.skim_rate, 2)
```

```python
        "data_disparity": signal["data_disparity"],
        "sentiment_bias": signal["sentiment_bias"],
        "rate_of_change": signal["rate_of_change"],
        "news_timing": signal["news_timing"],
        "structural": signal["structural"],
    }

    # Call Kimi K2.6 for diagnosis
    diagnosis = reasoning_engine.diagnose_trade(
        direction=trade["direction"],
        confidence=signal["confidence"],
        outcome=trade["outcome"],
        reasoning=signal["reasoning"] or "",
        signal_scores=signal_scores,
        market_title=market["title"] if market else trade["market_id"],
        price_at_entry=trade["price_paid"],
    )

    # Compute prediction error
    # Confidence was our estimate of winning. Actual was 0 (loss) or 1 (win).
    actual = 1.0 if trade["outcome"] == "WIN" else 0.0
    prediction_error = abs(signal["confidence"] - actual)

    # Store retro
    retro_id = insert_retro(conn, {
        "trade_id": trade_id,
        "market_id": trade["market_id"],
        "resolved_at": trade["resolved_at"] or now_iso(),
        "data_was_right": int(diagnosis.signal_accuracy.get("data_disparity", False)),
        "sentiment_was_right": int(diagnosis.signal_accuracy.get("sentiment_bias", False)),
        "roc_was_right": int(diagnosis.signal_accuracy.get("rate_of_change", False)),
        "news_was_right": int(diagnosis.signal_accuracy.get("news_timing", False)),
        "structural_was_right": int(diagnosis.signal_accuracy.get("structural", False)),
        "prediction_error": prediction_error,
        "lesson": diagnosis.lesson,
        "bias_detected": diagnosis.bias_detected,
        "weight_adjustment": None,  # computed below
        "retro_type": "normal",
        "category": category,
    })

    # Update weights based on signal accuracy
    weight_changes = update_weights(config.weights, diagnosis.signal_accuracy)
    if weight_changes:
        conn.execute(
            "UPDATE retros SET weight_adjustment = ? WHERE id = ?",
            (json.dumps(weight_changes), retro_id),
        )
        conn.commit()

    return retro_id


def update_weights(
    weights: SignalWeights,
    signal_accuracy: dict[str, bool],
) -> dict[str, float]:
    """
    Update signal weights using EMA based on accuracy.

    Signals that were right get weight increased, wrong get decreased.
    Returns dict of changes applied.
    """
    changes = {}
    signal_map = {
        "data_disparity": "data_disparity",
        "sentiment_bias": "sentiment_bias",
        "rate_of_change": "rate_of_change",
        "news_timing": "news_timing",
        "structural": "structural",
    }

    for signal_name, weight_attr in signal_map.items():
        was_right = signal_accuracy.get(signal_name)
        if was_right is None:
            continue

        current = getattr(weights, weight_attr)
        target = current * 1.1 if was_right else current * 0.9
        # EMA update
        new_weight = current + EMA_ALPHA * (target - current)
        new_weight = max(0.05, new_weight)  # floor at 5%

        change = new_weight - current
        changes[signal_name] = round(change, 6)
        setattr(weights, weight_attr, new_weight)

    # Re-normalize weights to sum to 1.0
    weights.normalize()

    return changes


def maybe_run_meta_retro(
    conn: sqlite3.Connection,
    reasoning_engine: ReasoningEngine,
    config: Config,
) -> int | None:
    """
    Run meta retro if we've hit the interval (every 20 resolved trades).
    Returns retro_id if a meta retro was run.
    """
    total_resolved = count_resolved_trades(conn)
    existing_metas = len(get_retros_by_type(conn, "meta"))

    # Run meta every N resolved trades
    expected_metas = total_resolved // config.meta_retro_interval
    if expected_metas <= existing_metas:
        return None

    # Collect recent retros
    recent = get_recent_retros(conn, limit=config.meta_retro_interval)
    if not recent:
        return None

    # Format summaries for Kimi K2.6
    summaries = []
    for r in recent:
        summaries.append(
            f"Trade {r['trade_id']}: "
            f"{'WIN' if r['prediction_error'] < 0.5 else 'LOSS'}, "
            f"error={r['prediction_error']:.2f}, "
            f"data={'right' if r['data_was_right'] else 'wrong'}, "
            f"sentiment={'right' if r['sentiment_was_right'] else 'wrong'}, "
            f"roc={'right' if r['roc_was_right'] else 'wrong'}, "
            f"structural={'right' if r['structural_was_right'] else 'wrong'}, "
            f"lesson: {r['lesson']}"
        )

    result = reasoning_engine.meta_retro(
        retro_summaries="\n".join(summaries),
        trade_count=len(recent),
    )

    retro_id = insert_retro(conn, {
        "trade_id": None,
        "market_id": None,
        "resolved_at": now_iso(),
        "data_was_right": None,
        "sentiment_was_right": None,
        "roc_was_right": None,
        "news_was_right": None,
        "structural_was_right": None,
        "prediction_error": None,
        "lesson": result.lesson,
        "bias_detected": result.bias_detected,
        "weight_adjustment": json.dumps(result.weight_recommendations) if result.weight_recommendations else None,
        "retro_type": "meta",
        "category": None,
    })

    # Apply weight recommendations if provided
    if result.weight_recommendations:
        for signal, delta in result.weight_recommendations.items():
            if hasattr(config.weights, signal):
                current = getattr(config.weights, signal)
                setattr(config.weights, signal, max(0.05, current + delta * EMA_ALPHA))
        config.weights.normalize()

    return retro_id


def run_wipe_retro(
    conn: sqlite3.Connection,
    reasoning_engine: ReasoningEngine,
    config: Config,
) -> int | None:
    """
    Run wipe retro when bankroll hits zero.
    Full post-mortem across all trades in the cycle.
    """
    bankroll = get_current_bankroll(conn, config.starting_bankroll)
    if bankroll > 0.01:
        return None

    all_trades = get_all_trades(conn)
    if not all_trades:
        return None

    # Format full history
    history_lines = []
    for t in all_trades:
        history_lines.append(
            f"Trade {t['id']}: {t['direction']} on {t['market_id']} "
            f"at {t['price_paid']}c x{t['contracts']}, "
            f"outcome={t['outcome']}, P/L=${t['profit_loss'] or 0:.2f}"
        )

    lesson = reasoning_engine.wipe_retro("\n".join(history_lines))

    retro_id = insert_retro(conn, {
        "trade_id": None,
        "market_id": None,
        "resolved_at": now_iso(),
        "data_was_right": None,
        "sentiment_was_right": None,
        "roc_was_right": None,
        "news_was_right": None,
        "structural_was_right": None,
        "prediction_error": None,
        "lesson": lesson,
        "bias_detected": "wipe",
        "weight_adjustment": None,
        "retro_type": "wipe",
        "category": None,
    })

    return retro_id
```

```python
# cipher/daemon.py
"""
Async daemon loop for continuous operation.

Polls Kalshi every 60 seconds, scores markets, pitches high-confidence trades,
auto-retros on resolution.
"""
from __future__ import annotations

import asyncio
import logging
import sqlite3
from datetime import datetime, timezone

import numpy as np

from cipher.candles import extract_features
from cipher.config import Config, ensure_data_dir
from cipher.db import (
    get_active_markets,
    get_candles,
    get_current_bankroll,
    get_market,
    get_open_trades,
    get_pending_pitch,
    get_portfolio_exposure,
    get_retros_for_category,
    get_retros_by_type,
    init_db,
    insert_candles,
    insert_signal,
    now_iso,
    upsert_market,
    count_denials_for_category,
)
from cipher.engine import SignalBundle, evaluate
from cipher.execute import check_resolutions, should_skim
from cipher.kalshi import KalshiClient
from cipher.pitch import (
    check_pitch_expiry,
    format_pitch,
    send_notification,
    send_pitch,
)
from cipher.reasoning import ReasoningEngine
from cipher.retro import maybe_run_meta_retro, run_trade_retro, run_wipe_retro
from cipher.signals.data_disparity import compute_data_disparity
from cipher.signals.news_timing import compute_news_timing
from cipher.signals.rate_of_change import compute_rate_of_change
from cipher.signals.sentiment import compute_sentiment
from cipher.signals.structural import compute_structural

logger = logging.getLogger("cipher")


async def run_daemon(config: Config) -> None:
    """Main daemon loop."""
    ensure_data_dir(config)
    conn = init_db(config.db_path)
    kalshi = KalshiClient(config)
    reasoning = ReasoningEngine(config.moonshot_api_key)

    logger.info("Cipher daemon started")
    send_notification(config.ntfy_topic, "Cipher Started", "Daemon is running.")

    try:
        while True:
            try:
                await _tick(conn, kalshi, reasoning, config)
            except Exception as e:
                logger.error(f"Tick error: {e}", exc_info=True)
            await asyncio.sleep(config.poll_interval_seconds)
    except KeyboardInterrupt:
        logger.info("Cipher daemon stopped")
    finally:
        kalshi.close()
        conn.close()


async def _tick(
    conn: sqlite3.Connection,
    kalshi: KalshiClient,
    reasoning: ReasoningEngine,
    config: Config,
) -> None:
    """One poll cycle."""
    # Check pitch expiry
    check_pitch_expiry(conn, config.pitch_expiry_minutes)

    # Don't scan if there's a pending pitch
    if get_pending_pitch(conn):
        logger.debug("Pending pitch exists, skipping scan")
        return

    # Check resolutions and run retros
    resolved = check_resolutions(conn, kalshi, config)
    for tid in resolved:
        run_trade_retro(conn, tid, reasoning, config)

    maybe_run_meta_retro(conn, reasoning, config)
    run_wipe_retro(conn, reasoning, config)

    # Profit skim check
    bankroll = get_current_bankroll(conn, config.starting_bankroll)
    skim_amount = should_skim(bankroll, config)
    if skim_amount > 0:
        logger.info(f"Skim ${skim_amount:.2f} from bankroll")
        send_notification(
            config.ntfy_topic, "Cipher: Skim",
            f"Skimming ${skim_amount:.2f}. Bankroll: ${bankroll:.2f} -> ${bankroll - skim_amount:.2f}",
        )

    # Poll markets
    try:
        markets = kalshi.get_active_markets()
    except Exception as e:
        logger.error(f"Failed to poll markets: {e}")
        return

    # Update market cache
    for m in markets:
        m["polled_at"] = now_iso()
        upsert_market(conn, m)

    # Filter: need sufficient volume
    candidates = [m for m in markets if (m.get("volume") or 0) >= 50]
    if not candidates:
        return

    # Score top candidates (limit to avoid excessive API calls)
    portfolio_exposure = get_portfolio_exposure(conn)
    open_positions = len(get_open_trades(conn))

    for m in candidates[:5]:
        decision = await _score_and_evaluate(
            conn, kalshi, reasoning, m, bankroll,
            portfolio_exposure, open_positions, config,
        )
        if decision and decision.action == "PITCH":
            signal_id = _store_and_pitch(conn, m, decision, bankroll, config)
            if signal_id:
                break  # One pitch at a time


async def _score_and_evaluate(
    conn: sqlite3.Connection,
    kalshi: KalshiClient,
    reasoning: ReasoningEngine,
    market: dict,
    bankroll: float,
    portfolio_exposure: float,
    open_positions: int,
    config: Config,
):
    """Score a single market and run through decision engine."""
    ticker = market["ticker"]

    # Fetch candles
    try:
        raw_candles = kalshi.get_candles(ticker)
        if raw_candles:
            insert_candles(conn, raw_candles)
    except Exception:
        pass

    candle_rows = get_candles(conn, market["id"], limit=60)
    if len(candle_rows) < 5:
        return None

    # Extract features
    closes = np.array([c["close"] for c in reversed(candle_rows)], dtype=np.float64)
    volumes = np.array([c["volume"] for c in reversed(candle_rows)], dtype=np.float64)

    close_time = market.get("close_time", "")
    hours_remaining = 24.0
    if close_time:
        try:
            ct = datetime.fromisoformat(close_time.replace("Z", "+00:00"))
            hours_remaining = max(0, (ct - datetime.now(timezone.utc)).total_seconds() / 3600)
        except ValueError:
            pass

    features = extract_features(
        closes, volumes,
        yes_bid=market.get("yes_bid") or 0,
        yes_ask=market.get("yes_ask") or 100,
        hours_remaining=hours_remaining,
    )

    # Compute time-series signals
    roc_result = compute_rate_of_change(closes, volumes)
    structural_result = compute_structural(
        last_price=market.get("last_price") or 50,
        yes_bid=market.get("yes_bid"),
        yes_ask=market.get("yes_ask"),
        volume=market.get("volume") or 0,
        hours_remaining=hours_remaining,
        recent_volume=int(np.sum(volumes[-5:])) if len(volumes) >= 5 else 0,
    )

    # Build candle summary for LLM
    candle_summary = (
        f"Momentum: {features.momentum:.4f}, "
        f"Volume spike: {features.volume_spike:.2f}x, "
        f"Volatility: {features.volatility:.4f}, "
        f"Mean reversion: {features.mean_reversion:.4f}, "
        f"Spread: {features.bid_ask_spread}c, "
        f"N candles: {features.n_candles}"
    )

    # Get retro lessons for this category
    category = market.get("category", "unknown")
    category_retros = get_retros_for_category(conn, category)
    relevant_retros = "\n".join(
        f"- {r['lesson']}" for r in category_retros if r["lesson"]
    )
    meta_retros = get_retros_by_type(conn, "meta")
    meta_lessons = "\n".join(
        f"- {r['lesson']}" for r in meta_retros[:3] if r["lesson"]
    )

    # Call Kimi K2.6 for LLM signals
    try:
        llm_result = reasoning.score_market(
            title=market.get("title", ticker),
            yes_price=market.get("last_price") or 50,
            close_time=close_time,
            hours_remaining=hours_remaining,
            category=category,
            candle_summary=candle_summary,
            roc_score=roc_result.score,
            roc_reasoning=f"velocity={roc_result.velocity:.4f}, accel={roc_result.acceleration:.4f}",
            structural_score=structural_result.score,
            structural_reasoning=f"decay={structural_result.time_decay_distortion:.2f}, liquidity={structural_result.liquidity_score:.2f}",
            relevant_retros=relevant_retros,
            meta_lessons=meta_lessons,
        )
    except Exception as e:
        logger.error(f"LLM scoring failed for {ticker}: {e}")
        return None

    # Compute LLM-driven signals
    data_disp = compute_data_disparity(
        market_price=market.get("last_price") or 50,
        llm_estimated_probability=llm_result.true_probability,
        llm_confidence=llm_result.confidence,
    )
    sentiment = compute_sentiment(
        llm_sentiment_score=llm_result.sentiment_bias_score,
        llm_sentiment_confidence=llm_result.sentiment_bias_confidence,
        bias_flags=llm_result.bias_flags,
    )
    news = compute_news_timing()

    # Build signal bundle
    signals = SignalBundle(
        data_disparity=data_disp.score,
        data_disparity_conf=data_disp.confidence,
        sentiment_bias=sentiment.score,
        sentiment_bias_conf=sentiment.confidence,
        rate_of_change=roc_result.score,
        rate_of_change_conf=roc_result.confidence,
        news_timing=news.score,
        news_timing_conf=news.confidence,
        structural=structural_result.score,
        structural_conf=structural_result.confidence,
    )

    category_denials = count_denials_for_category(conn, category)

    return evaluate(
        signals=signals,
        market_price=market.get("last_price") or 50,
        estimated_prob=llm_result.true_probability,
        bankroll=bankroll,
        portfolio_exposure=portfolio_exposure,
        open_positions=open_positions,
        category=category,
        config=config,
        reasoning=llm_result.reasoning,
        category_denial_count=category_denials,
    )


def _store_and_pitch(
    conn: sqlite3.Connection,
    market: dict,
    decision,
    bankroll: float,
    config: Config,
) -> int | None:
    """Store signal and send pitch notification."""
    signal_id = insert_signal(conn, {
        "market_id": market["id"],
        "timestamp": now_iso(),
        "data_disparity": decision.combined_score,  # simplified for storage
        "data_disparity_conf": decision.confidence,
        "sentiment_bias": None,
        "sentiment_bias_conf": None,
        "rate_of_change": None,
        "rate_of_change_conf": None,
        "news_timing": None,
        "news_timing_conf": None,
        "structural": None,
        "structural_conf": None,
        "combined_score": decision.combined_score,
        "confidence": decision.confidence,
        "direction": decision.direction,
        "reasoning": decision.reasoning,
        "action": "PITCH",
    })

    msg = format_pitch(
        decision=decision,
        market_title=market.get("title", market["ticker"]),
        market_price=market.get("last_price") or 50,
        estimated_prob=decision.confidence * 100,
        bankroll=bankroll,
        signal_scores={},
    )

    send_pitch(config.ntfy_topic, msg, signal_id)
    logger.info(f"Pitched {market['ticker']} ({decision.direction}, conf={decision.confidence:.2f})")
    return signal_id
```
